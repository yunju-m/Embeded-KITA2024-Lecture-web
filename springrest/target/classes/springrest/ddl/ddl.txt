-- RESTBOARD 계정 생성
CREATE USER RESTBOARD IDENTIFIED BY RESTBOARD;
GRANT CONNECT, RESOURCE TO RESTBOARD;

-- RESTBOARD 테이블 생성
CREATE TABLE RESTBOARD(
	RBID NUMBER PRIMARY KEY,
	RBWRITER VARCHAR2(200) NOT NULL,
	RBTITLE VARCHAR2(200) NOT NULL,
	RBCONTENT VARCHAR2(4000),
	RBREGDATE TIMESTAMP
);

-- RESTBOARD 시퀀스 생성
CREATE SEQUENCE SEQ_RESTBOARD;

SELECT * FROM RESTBOARD;
